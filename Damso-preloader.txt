<script>
(function() {
	const preloader = document.getElementById('dd-preloader');
	
	// SESSION STORAGE CHECK - Prevents annoyance on every page load
	const hasSeenPreloader = sessionStorage.getItem('dd_preloader_seen');
	if (hasSeenPreloader === 'true' && !window.Shopify.designMode) {
		preloader.style.display = 'none';
document.body.classList.remove('is-loading');
setTimeout(() => document.dispatchEvent(new CustomEvent('DD_Engines_Ready')), 100);
return; 
	}
	
	const interactiveCounter = document.getElementById('interactive-counter');
	const percentEl = document.getElementById('load-percent');
	const uiContainer = document.querySelector('.preloader-ui');
	const video = document.getElementById('preloader-hidden-video');
	const canvas = document.getElementById('webgl-preloader-canvas');
	const sfx = document.getElementById('dd-glitch-sfx');
	
	document.body.classList.add('is-loading');
	
	// WebGL variables scoped outside so they can be securely disposed
	let renderer, scene, camera, shaderMaterial, geometry, plane, animationId;
	let isPreloaderActive = true; 
	
	// Initialize WebGL only AFTER video is confirmed playing
	const initWebGL = () => {
		renderer = new THREE.WebGLRenderer({ canvas, antialias: true, alpha: true, powerPreference: "high-performance" });
		renderer.setSize(window.innerWidth, window.innerHeight);
		renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
		
		scene = new THREE.Scene();
		const aspect = window.innerWidth / window.innerHeight;
		camera = new THREE.OrthographicCamera(-aspect, aspect, 1, -1, 0, 1);
		
		const videoTexture = new THREE.VideoTexture(video);
		videoTexture.minFilter = THREE.LinearFilter;
		videoTexture.magFilter = THREE.LinearFilter;
		videoTexture.format = THREE.RGBAFormat;
		
		// FIXED SHADER: Valid GLSL syntax preventing silent crashes
		shaderMaterial = new THREE.ShaderMaterial({
			uniforms: {
				tDiffuse: { value: videoTexture },
				uTime: { value: 0.0 },
				uGlitch: { value: 0.0 } 
			},
			vertexShader: `
			varying vec2 vUv;
			void main() {
				vUv = uv;
				gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
			}
			`,
			fragmentShader: `
			uniform sampler2D tDiffuse;
			uniform float uTime;
			uniform float uGlitch;
			varying vec2 vUv;
			
			// Proper random function declaration
			float random(vec2 st) {
				return fract(sin(dot(st.xy, vec2(12.9898,78.233))) * 43758.5453123);
			}
			
			void main() {
				vec2 uv = vUv;
				
				// Glitch tearing logic
				float blocks = floor(uv.y * 30.0);
				float tear = (random(vec2(blocks, uTime)) - 0.5) * uGlitch * 0.15;
				uv.x += tear;
				
				// RGB Split logic
				float split = uGlitch * 0.06;
				float r = texture2D(tDiffuse, clamp(uv + vec2(split, 0.0), 0.0, 1.0)).r;
				float g = texture2D(tDiffuse, clamp(uv, 0.0, 1.0)).g;
				float b = texture2D(tDiffuse, clamp(uv - vec2(split, 0.0), 0.0, 1.0)).b;
				
				vec3 color = vec3(r, g, b);
				
				// Vignette mask
				float dist = distance(vUv, vec2(0.5));
				float mask = smoothstep(0.8, 0.2, dist);
				
				float alpha = (1.0 - uGlitch) * mask;
				
				gl_FragColor = vec4(color, alpha);
			}
			`,
			transparent: true
		});
		
		geometry = new THREE.PlaneGeometry(2, 2);
		plane = new THREE.Mesh(geometry, shaderMaterial);
		scene.add(plane);
		
		let vidAspect = video.videoWidth / video.videoHeight || 16/9; 
		
		const resizeVideoPlane = () => {
			if(!isPreloaderActive) return;
			const screenAspect = window.innerWidth / window.innerHeight;
			
			camera.left = -screenAspect;
			camera.right = screenAspect;
			camera.top = 1;
			camera.bottom = -1;
			camera.updateProjectionMatrix();
			
			// Maintain video aspect ratio while scaling
			let scale = 0.8; 
			if (screenAspect < vidAspect) {
				scale = (screenAspect / vidAspect) * 0.9; 
			}
			
			plane.scale.set(vidAspect * scale, scale, 1);
		};
		
		window.addEventListener('resize', () => {
			if(!isPreloaderActive) return;
			renderer.setSize(window.innerWidth, window.innerHeight);
			resizeVideoPlane();
		});
		
		resizeVideoPlane();
		
		const clock = new THREE.Clock();
		
		const animateWebGL = () => {
			if (!isPreloaderActive) return; 
			animationId = requestAnimationFrame(animateWebGL);
			shaderMaterial.uniforms.uTime.value = clock.getElapsedTime();
			renderer.render(scene, camera);
		};
		animateWebGL();
	};
	
	// Ensure video is playing before firing WebGL to prevent texture errors
	let isVideoPlaying = false;
	video.play().then(() => {
		isVideoPlaying = true;
		initWebGL();
	}).catch(e => {
		console.log("Video autoplay blocked by browser policy, waiting for interaction.");
		// Fallback if autoplay is blocked
		initWebGL(); 
	});
	
	
	// --- LOADING SEQUENCE ---
	let loadProgress = { val: 0 };
	let isReadyToEnter = false;
	let hasEntered = false;
	
	// Check if fonts are ready for smoother rendering
	if(document.fonts) {
		document.fonts.ready.then(() => startLoading());
	} else {
		setTimeout(startLoading, 200);
	}
	
	function startLoading() {
		gsap.to(uiContainer, { opacity: 1, duration: 0.5, ease: "power2.out" });
		
		gsap.to(loadProgress, {
			val: 100,
		  duration: 3.0, // Slightly faster load for better UX
		  ease: "power2.inOut",
		  onUpdate: () => {
			  if(percentEl) percentEl.innerText = Math.floor(loadProgress.val).toString().padStart(2, '0');
		  },
		  onComplete: () => {
			  showEnterState();
		  }
		});
	}
	
	const showEnterState = () => {
		if (isReadyToEnter) return;
		isReadyToEnter = true;
		
		gsap.to(interactiveCounter, { opacity: 0, duration: 0.3, onComplete: () => {
			interactiveCounter.innerHTML = "[ ENTER ]";
			interactiveCounter.classList.add('enter-state');
			gsap.to(interactiveCounter, { opacity: 1, duration: 0.5 });
		}});
		
		// Use pointerdown for instant mobile response
		interactiveCounter.addEventListener('pointerdown', triggerExitSequence);
	};
	
	// --- EXIT & DISPOSAL SEQUENCE ---
	const triggerExitSequence = () => {
		if (hasEntered) return;
		hasEntered = true;
		
		sessionStorage.setItem('dd_preloader_seen', 'true');
		
		if (sfx) {
			sfx.volume = 0.8;
			sfx.play().catch(e => console.log("Audio blocked"));
		}
		
		if (window.gsap && shaderMaterial) {
			const tl = gsap.timeline({
				onComplete: () => {
					// 1. Hide DOM Elements
					preloader.style.display = 'none';
			document.body.classList.remove('is-loading');
			
			// 2. Dispatch event to start the rest of the site
			document.dispatchEvent(new CustomEvent('DD_Engines_Ready'));
			
			// 3. CRITICAL PERFORMANCE FIX: Nuke WebGL from memory
			isPreloaderActive = false; 
			cancelAnimationFrame(animationId);
			
			if(geometry) geometry.dispose();
			if(shaderMaterial) {
				shaderMaterial.uniforms.tDiffuse.value.dispose(); // Dispose video texture
				shaderMaterial.dispose();
			}
			if(renderer) {
				renderer.dispose();
				renderer.forceContextLoss(); // Hard flush GPU
			}
				}
			});
			
			// The dramatic glitch out
			tl.to(shaderMaterial.uniforms.uGlitch, { value: 1.0, duration: 1.0, ease: "power3.in" }, 0)
			.to(uiContainer, { opacity: 0, scale: 1.1, duration: 0.5, ease: "power2.in" }, 0)
			.to(preloader, { opacity: 0, duration: 0.6, ease: "none" }, 0.5);
		} else {
			// Fallback if GSAP fails
			preloader.style.display = 'none';
document.body.classList.remove('is-loading');
		}
	};
	
})();
</script>
